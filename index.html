from datetime import date
import asyncio
from pyscript import document

async def type_writer(text, delay=0.8):
    output = document.getElementById("output")
    output.innerText = ""  # start empty

    for line in text.strip().split("\n"):
        output.innerText += line + "\n"
        await asyncio.sleep(delay)

async def greet():
    target_date = date(2026, 4, 12)
    today = date.today()

    poem = """Like morning light that softly breaks,
You came into my life awake.
A quiet peace, a gentle grace,
A love I never can replace.

In every smile, in every sigh,
You are my reason, my “good try.”
Through storms or calm, near or far,
You are my constant guiding star.

Jenniva, my heart’s sweet song,
With you is where I belong.
Not just today, but all my days,
I’ll love you in a thousand ways. ☺️❤️
"""

    special_poem = """Today the world feels warm and bright,
Because our love stands pure and right.
A journey shared, a bond so true,
And every moment leads to you.

Happy Anniversary, my love Jenniva ❤️
You are my answered prayer.
"""

    message = special_poem + "\n\n" + poem if today == target_date else poem
    await type_writer(message, 0.9)

asyncio.ensure_future(greet())
